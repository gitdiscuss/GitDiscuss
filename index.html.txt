<!-- DYNAMIC GISCUS LOADER SCRIPT                            -->

<script>
  // 1. Map your Giscus Categories to their actual Category IDs on GitHub.
  // Replace the placeholder IDs below with the IDs from your repository!
  const categories = {
    general: { name: "General", id: "DIC_kwDOTZeSG84DBQyA" }, // <-- Already configured!
    announcements: { name: "Announcements", id: "DIC_kwDOTZeSG84DBQx-" },
    qa: { name: "Q&A", id: "DIC_kwDOTZeSG84DBQyC" },
    ideas: { name: "Ideas", id: "DIC_kwDOTZeSG84DBQyC" },
    polls: { name: "Polls", id: "DIC_kwDOTZeSG84DBQyE" },
    showAndTell: { name: "Show and Tell", id: "DIC_kwDOTZeSG84DBQyD" }
  };

  // 2. The core loading function
  function loadGiscus(categoryKey) {
    const categoryObj = categories[categoryKey];
    if (!categoryObj) return;

    const container = document.querySelector('.giscus');
    container.innerHTML = ''; // Wipe out the previous iframe

    // Build the script element from scratch
    const script = document.createElement('script');
    script.src = "https://giscus.app/client.js";
    script.setAttribute('data-repo', "treatwashere/GitDiscuss");
    script.setAttribute('data-repo-id', "R_kgDOTZeSGw");
    script.setAttribute('data-category', categoryObj.name);
    script.setAttribute('data-category-id', categoryObj.id);
    script.setAttribute('data-mapping', "pathname");
    script.setAttribute('data-strict', "0");
    script.setAttribute('data-reactions-enabled', "1");
    script.setAttribute('data-emit-metadata', "0");
    script.setAttribute('data-input-position', "bottom");
    script.setAttribute('data-theme', "dark_tritanopia");
    script.setAttribute('data-lang', "en");
    script.setAttribute('crossorigin', "anonymous");
    script.async = true;

    // Inject into the page
    container.appendChild(script);
  }

  // 3. Event Listener: Swapping when dropdown option changes
  document.getElementById('topic-select').addEventListener('change', (e) => {
    loadGiscus(e.target.value);
  });

  // 4. Load the default option (General) on initial page load
  loadGiscus('general');
</script>
